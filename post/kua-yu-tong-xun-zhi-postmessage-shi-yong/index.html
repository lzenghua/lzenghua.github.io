跨域通讯之postMessage使用

  1. 2020-09-23
  2. 277字
  3. 2 min read

在sdk业务中，常常会出现如下场景：1. 需要与客户端进行信息的传递；2. 发行中嵌套iframe的跨域通信等等。

一、父窗口向子窗口通信

1. 父窗口发送信息

<!-- 嵌套子窗口的iframe；需要定义一个id -->
<iframe id="child_iframe" src="./iframe.html"></iframe>

var data = {
    type: "ddth5sdk",
    action: "xxx"
};
document.getElementById('child_iframe').contentWindow.postMessage(JSON.stringify(data), "*");

2. 子窗口监听信息

window.addEventListener('message', function (event) {
    if (event.data && typeof event.data === 'string') {
        var data = JSON.parse(event.data);
        if (data && data.type === 'ddth5sdk') {
            // 收到父窗口的信息，在此处理数据
        }
    }
});

二、子窗口向父窗口通信

1. 子窗口发送信息

var data = {
    action: "ddtsdk",
    action_type: "login"
};
window.parent.postMessage(JSON.stringify(data), "*");
// 如果是向顶层发消息，则使用下面的代码
window.top.postMessage(JSON.stringify(data), "*");

2. 父窗口监听信息

window.addEventListener('message', function (event) {
        if (event.data && typeof event.data === 'string') {
            var data = JSON.parse(event.data);
            if (data && data.action === 'ddtsdk' && data.action_type == "login") {
                // 收到子窗口的信息，在此处理数据
            }
        }
    });
